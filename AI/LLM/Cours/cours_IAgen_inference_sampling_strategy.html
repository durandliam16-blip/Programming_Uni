def decode(list_jetons):
    assert isinstance(list_jetons, list), f"{type(list_jetons)}"
    phrase_sortie = ""
    for tid in list_jetons:
        phrase_sortie += tokenizer.decode([tid])
    return phrase_sortie

longeur_sortie = 10  # à changer à 5
nb_faisceaux = 2
generated = inputs['input_ids']  # (nb_faisceau, longeur_entree) ou (1, longeur_entree) à la 1e itération
faisceaux_total_prob = None
histoire_faisceaux_total_prob = np.zeros((nb_faisceaux, longeur_sortie))

for idx_jeton_genere in range(longeur_sortie):
    print("=========================================\n")
    print(f"itération {idx_jeton_genere}")
    print("Les faiceaux d'entrée :")
    for idx_phrase_entre in range(generated.shape[0]):
        print(decode(generated[idx_phrase_entre].cpu().tolist()))
    print("------------\n")
    
    with torch.no_grad():
        outputs = model(input_ids=generated)  # generated: (nb_faisceaux, longeur_entree à cette itération)

    logits = outputs.logits  # (nb_faisceau, seq_len, vocab_size)
    jeton_suivant_prob = torch.nn.functional.softmax(logits[:, -1, :], dim=1)  # (nb_faisceaux, vocab_size)
    
    if faisceaux_total_prob is None:
        # 1e itération
        # garde "nb_faiceau" jetons
        top_prob, top_jetons = torch.topk(jeton_suivant_prob.reshape(-1), k=nb_faisceaux)     
        # top_logits: (nb_faisceaux,)
        # top_jetons: (nb_faisceaux,)
        
        faisceaux_total_prob = top_prob.reshape(-1, 1)  # (nb_faisceaux, 1) 
        
        # ajoute top_jetons dans la phrase de sortie
        # car il y a "nb_faisceaux" jetons à conserver à la 1e itération, il faut multiplier la phrase de sorties "nb_faisceaux" fois
        # pour représenter "nb_faisceaux" options de la phrase de sortie
        generated = torch.cat([generated.repeat(nb_faisceaux, 1), top_jetons.reshape(-1, 1)], dim=1)  # (nb_faisceaux, longeur_entree + 1)

    else:
        # à partir de la 2e itération
        # montrer topk de chaque faisceau d'entrée
        for idx_phrase_entre in range(generated.shape[0]):
            print("===")
            _prob_entree = float(faisceaux_total_prob[idx_phrase_entre])
            print("Phrase d'entrée: ", decode(generated[idx_phrase_entre].cpu().tolist()), f"\t prob: {_prob_entree}")
            print("Top jetons suivant \t\t Prob jeton \t\t Prob phrase sorite")
            _top_prob, _top_jetons = torch.topk(jeton_suivant_prob[idx_phrase_entre], k=5)
            
            _top_jetons = _top_jetons.cpu().tolist()
            for _idx_jeton in range(_top_prob.shape[0]):
                _prob_jeton = np.round(float(_top_prob[_idx_jeton]), decimals=4)
                _prob_phrase_et_jeton = np.round(_prob_jeton * _prob_entree, decimals=4)
                print(f"{tokenizer.decode([_top_jetons[_idx_jeton]])} \t\t\t\t {_prob_jeton} \t\t\t\t {_prob_phrase_et_jeton}")
        
        """
        mise à jour les faiceaux et leur total logit 
        """
        jeton_suivant_prob = jeton_suivant_prob *  faisceaux_total_prob  # (nb_faisceaux, vocab_size)
        top_prob, top_jetons = torch.topk(jeton_suivant_prob.reshape(-1), k=nb_faisceaux)  # top_logits : (nb_faisceaux,)
        # convertir les indices aplati en index 2D pour défaire reshape(-1) appliqué au jeton_suivant_logits
        vocab_size = jeton_suivant_logits.shape[-1]
        ind_top_faiceaux = top_jetons // vocab_size  # (nb_faisceaux,)
        top_jetons = top_jetons % vocab_size  # (nb_faisceaux,)
        
        # faiceaux sélectionés
        generated = generated[ind_top_faiceaux]  # (nb_faisceaux, longeur_entree)
        # ajoute top_jetons aux faiceaux sélectionés
        generated = torch.cat([generated, top_jetons.reshape(-1, 1)], dim=1)
        
        faisceaux_total_prob = top_prob.reshape(nb_faisceaux, -1)  # (nb_faisceaux, 1)
    
    print("Les faiceaux d'sortie :")
    print("Phrase \t Prob")
    for idx_phrase_sorite in range(generated.shape[0]):
        print(f"{decode(generated[idx_phrase_sorite].cpu().tolist())} \t {float(faisceaux_total_prob[idx_phrase_sorite])}")

    histoire_faisceaux_total_prob[:, idx_jeton_genere] += faisceaux_total_prob.reshape(-1).cpu().numpy()

def recherche_dichotonomique(tab, ele):
    # return idx of the ele that is less or equal to ele
    begin = 0
    fin = len(tab) - 1
    found = False
    mid = -1
    while not found and begin <= fin:
        mid = (begin + fin) // 2
        if tab[mid] == ele:
            found = True
        elif tab[mid] < ele:
            begin = mid + 1
        else:
            fin = mid - 1  # fin peut arriver à -1

    if found:
        return mid
    else:
        # return ele that is less than ele
        # print(f"tab[begin]: {tab[begin]} | {ele}")
        # print(f"tab[fin]: {tab[fin]} | {ele}")
        # print("===")
        return begin
            
        
def sampling_categories(prob_cumulative) -> int:
    u = np.random.rand()
    return recherche_dichotonomique(prob_cumulative, u)

longeur_sortie = 25
keep_topk = 10
val_top_p = 0.95
generated = inputs['input_ids']  # (1, nb_jeton_entree)
vocab_size = None
all_ids = None

for i in tqdm(range(longeur_sortie)):
    with torch.no_grad():
        outputs = model(input_ids=generated)
        
    logits = outputs.logits  # (batch_size, seq_len, vocab_size)
    jeton_suivant_logits = logits[:, -1, :]  # (batch_size, vocab_size)

    if vocab_size is None:
        vocab_size = jeton_suivant_logits.shape[-1]
        all_ids = np.arange(vocab_size)

    # échantillonnage
    jeton_suivant_prob = torch.nn.functional.softmax(jeton_suivant_logits[0])  # (vocab_size,)
    sorted_indices = torch.argsort(jeton_suivant_prob,  descending=True)
    
    somme_prob = 0
    idx = 0
    top_indices = list()
    top_prob = list()
    while somme_prob < val_top_p:
        chosen_idx = sorted_indices[idx]
        somme_prob += float(jeton_suivant_prob[chosen_idx])

        top_indices.append(int(chosen_idx))
        top_prob.append(float(jeton_suivant_prob[chosen_idx]))
        # ---
        idx += 1

    top_prob = np.array(top_prob) / val_top_p
    top_prob_cumulative = np.cumsum(top_prob)
    
    sampled_index = sampling_categories(top_prob_cumulative)
    jeton_suivant = torch.from_numpy(np.array([top_indices[sampled_index]])).to(generated.device).reshape(1, 1)
    generated = torch.cat([generated, jeton_suivant], dim=1)
